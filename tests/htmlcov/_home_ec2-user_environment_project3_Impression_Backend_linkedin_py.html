17#### Exchange auth code for access token 

18def get_access_token(auth_code): 

19 base_url = "https://www.linkedin.com/oauth/v2/accessToken" 

20 query = "?grant_type=authorization_code&code=%s&redirect_uri=%s&client_id=%s&client_secret=%s" % (auth_code, REDIRECT_URI, CLIENT_ID, CLIENT_SECRET) 

21 response = requests.get(base_url + query) 

22 data = json.loads(response.text) 

23 return data["access_token"]
